import tkinter import math import random import time tk = tkinter.Tk() tk.withdraw() sw = tk.winfo_screenwidth() sh = tk.winfo_screenheight() words = ["520","喜欢你","偏爱你","岁岁年年","满心是你"] color = ["#ff8da1","#ffb3ba","#ffc8dd","#ffd6e0"] heart = [] for i in range(60): t = i * 0.105 x = 16 * math.sin(t)**3 y = 13*math.cos(t)-5*math.cos(2*t)-2*math.cos(3*t)-math.cos(4*t) heart.append((x*20, -y*20)) print("即将开始 3 秒倒计时") for s in range(3,0,-1): print(f"倒计时 {s} 秒") time.sleep(1) print("爱心弹窗启动！") for x,y in heart: w = tkinter.Toplevel() w.geometry(f"90x40+{int(sw/2+x)}+{int(sh/2+y-80)}") w.config(bg=random.choice(color)) t = tkinter.Label(w,text=random.choice(words),font=("黑体",12),fg="white") t.pack() w.update() time.sleep(0.03) tk.mainloop()
