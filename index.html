import tkinter as tk import math import random import time # ========== 自定义设置 ========== text_list = ["爱", "你", "永", "远", "❤️", "开", "心", "每", "一", "天"] win_width = 110 win_height = 45 colors = ['#FFB6C1', '#FFC0CB', '#FF69B4', '#FF1493', '#DB7093', '#FFA07A'] # ============================== def generate_heart_points(num_points=80): points = [] for i in range(num_points): t = i * (2 * math.pi) / num_points x = 16 * math.sin(t)**3 y = 13 * math.cos(t) - 5 * math.cos(2*t) - 2 * math.cos(3*t) - math.cos(4*t) scale = 22 points.append((x * scale, -y * scale)) return points def create_heart_windows(): root = tk.Tk() root.withdraw() screen_w = root.winfo_screenwidth() screen_h = root.winfo_screenheight() windows = [] points = generate_heart_points(80) for (x, y) in points: txt = random.choice(text_list) c = random.choice(colors) win = tk.Toplevel() win.geometry(f"{win_width}x{win_height}+{int(screen_w/2 + x)}+{int(screen_h/2 + y - 100)}") win.config(bg=c) lbl = tk.Label(win, text=txt, bg=c, fg="white", font=("微软雅黑", 11, "bold")) lbl.pack(expand=True) windows.append(win) win.update() time.sleep(0.02) root.mainloop() # 关键修复：让窗口不闪退 if __name__ == "__main__": print("即将开始弹窗，倒计时3秒...") for i in range(3, 0, -1): print(f"倒计时：{i} 秒") time.sleep(1) print("开始啦！") create_heart_windows()
